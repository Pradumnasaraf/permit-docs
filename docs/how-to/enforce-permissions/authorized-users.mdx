To get all users who have access to a resource, you can use the `permit.GetAuthorizedUsers` function.
This function returns a list of users who have access to perform a specific action on a resource.

## Simple Usage

The `permit.GetAuthorizedUsers` function accepts an action and a Resource as input,
and returns a list of users who have access to perform the given action on the given resource,
including the user's attributes:

<Tabs groupId="language">
<TabItem value="nodejs" label="Node.js">

```js
const { Permit } = require("permitio");

const permit = new Permit({token: "<YOUR_API_KEY>", ...});
const allowedUsers = await permit.getAuthorizedUsers("push",{type: "repo", id: "OPAL"});
```

</TabItem>

<TabItem value="golang" label="GoLang">

```go
package main

import (
	"encoding/json"
	"fmt"
)
import p "github.com/permitio/permit-golang/pkg/permit"
import "github.com/permitio/permit-golang/pkg/config"
import "github.com/permitio/permit-golang/pkg/enforcement"

func main() {
	// Create permit client
	permitConfig := config.NewConfigBuilder("<YOUR_API_TOKEN>").Build()
	permit := p.New(permitConfig)

	// Create user and resource variables
	resource := enforcement.ResourceBuilder("repo").WithID("OPAL").Build()

	var userPermissions enforcement.UserPermissions
	// List user permissions
	allowedUsers, err := permit.GetAuthorizedUsers(
		"push",
		resource
    )
	if err != nil {
		fmt.Printf("Error getting authorized users: %s", err)
	} else if len(allowedUsers) > 0 {
		fmt.Println("There are users who are allowed to perform the action")
		for user := range allowedUsers {
		    fmt.Printf("User '%s' is allowed to perform the given action on the given resource",user)
		}
	} else {
		fmt.Println("There are no users who are allowed to perform the action")
	}
}

```
